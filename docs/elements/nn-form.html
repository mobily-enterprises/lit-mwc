          ? (el.value ? el.value : 'on')
          : undefined

        return el[valueSource]
      } else {
        return el[valueSource]

    } else {
      const nonRadio = elements.filter(el => !this._radioElement(el))

        console.error('Duplicate name', elName, 'for non-radio elements')
